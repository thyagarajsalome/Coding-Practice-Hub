1. Basic HTTP Server (Easy)

Problem: Create a simple HTTP server using Node.jsâ€™s built-in http module. The server should listen on a specific port and respond with a plain text message. 


// File: server.js
const http = require('http');
const port = 3000;

const server = http.createServer((req, res) => {
  res.writeHead(200, { 'Content-Type': 'text/plain' });
  res.end('Hello from Node.js HTTP server!');
});

server.listen(port, () => {
  console.log(`Server running at http://localhost:${port}/`);
});

2. Asynchronous File Reading (Slightly Medium)

Problem: Read a file asynchronously using the fs module. Log its content to the console and handle any errors. 


// File: readFile.js
const fs = require('fs');

fs.readFile('example.txt', 'utf8', (err, data) => {
  if (err) {
    return console.error('Error reading file:', err);
  }
  console.log('File content:', data);
});

3. Basic Express Server (Medium)

Problem: Build a simple Express server that responds to GET requests on the root route (/) with a welcome message. 


// File: expressServer.js
const express = require('express');
const app = express();
const port = 3000;

app.get('/', (req, res) => {
  res.send('Hello from Express!');
});

app.listen(port, () => {
  console.log(`Express server listening on http://localhost:${port}`);
});

4. Express Middleware (Medium-Hard)

Problem: Implement a custom middleware function in Express to log the HTTP method and URL of each request. Then, create a basic route to demonstrate the middleware in action. 


// File: middlewareExample.js
const express = require('express');
const app = express();
const port = 3000;

// Custom middleware to log request details
app.use((req, res, next) => {
  console.log(`${req.method} ${req.url}`);
  next();
});

app.get('/', (req, res) => {
  res.send('Check the console for logged requests.');
});

app.listen(port, () => {
  console.log(`Server with middleware running at http://localhost:${port}`);
});

5. CLI Application with Async/Await (Hard)

Problem: Create a command-line application that accepts a filename as an argument and reads its content asynchronously using async/await. Log the file content or an error message if the file does not exist. 


// File: cliApp.js
const fs = require('fs').promises;

async function main() {
  try {
    const filename = process.argv[2];
    if (!filename) {
      console.error('Please provide a filename as an argument.');
      process.exit(1);
    }
    const data = await fs.readFile(filename, 'utf8');
    console.log(`Content of ${filename}:`);
    console.log(data);
  } catch (error) {
    console.error('Error:', error);
  }
}

main();
