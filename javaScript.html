Problem 5 (Hard): Deep Copying an Object

Problem: Write a function that creates a deep copy of a nested object (without using JSON methods). 


function deepCopy(obj) {
  if (obj === null || typeof obj !== "object") {
    return obj;
  }
  // Handle Array
  if (Array.isArray(obj)) {
    return obj.map(item => deepCopy(item));
  }
  // Handle Object
  const copy = {};
  for (let key in obj) {
    if (obj.hasOwnProperty(key)) {
      copy[key] = deepCopy(obj[key]);
    }
  }
  return copy;
}

// Example
const original = {
  name: "Alice",
  details: {
    age: 25,
    hobbies: ["reading", "gaming"]
  }
};

const copied = deepCopy(original);
copied.details.age = 30;
console.log("Original Age:", original.details.age);
console.log("Copied Age:", copied.details.age);

    return fn.apply(context, args.concat(newArgs));
  };

Problem 3 (Medium): Simple Publish/Subscribe Pattern

Problem: Implement a simple pub/sub system in JavaScript. 


class EventEmitter {
  constructor() {
    this.events = {};
  }
  
  subscribe(event, listener) {
    if (!this.events[event]) {
      this.events[event] = [];
    }
    this.events[event].push(listener);
  }
  
  publish(event, data) {
    if (this.events[event]) {
      this.events[event].forEach(listener => listener(data));
    }
  }
}

// Usage example:
const emitter = new EventEmitter();
emitter.subscribe("message", data => console.log("Received:", data));
emitter.publish("message", "Hello, pub/sub!");
